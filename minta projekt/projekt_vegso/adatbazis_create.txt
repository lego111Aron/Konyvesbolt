CREATE TABLE MUFAJ
(
    "ID" NUMBER NOT NULL ENABLE,
    "MUFAJ_NEV" VARCHAR2(40 BYTE) NOT NULL ENABLE,
    "ALMUFAJ_NEV" VARCHAR2(40 BYTE) NOT NULL ENABLE,

CREATE TABLE SZERZO

    "ID" NUMBER NOT NULL ENABLE,
    "NEV" VARCHAR2(40 BYTE) NOT NULL ENABLE,

    PRIMARY KEY ("ID") ENABLE
);


CREATE TABLE KONYV
(
    "ID" NUMBER NOT NULL ENABLE, 
    "CIM" VARCHAR2(50 BYTE), 
    "SZERZO_ID" NUMBER, 
    "KIADO_ID" NUMBER, 
    "OLDALSZAM" NUMBER,
    "LEIRAS" VARCHAR2(256 BYTE),
    "NYELV" VARCHAR2(20 BYTE),
    "MUFAJ_ID" NUMBER,
    "AR" NUMBER,
    "KIFUTO" NUMBER(1,0) DEFAULT 0 NOT NULL,

    PRIMARY KEY ("ID") ENABLE,

    CONSTRAINT KONYV_SZERZOFK
        FOREIGN KEY (SZERZO_ID)
        REFERENCES SZERZO(ID)
        ON DELETE SET NULL,
    CONSTRAINT KONYV_KIADOFK
        FOREIGN KEY (KIADO_ID)
        REFERENCES KIADO(ID)
        ON DELETE SET NULL,
    CONSTRAINT KONYV_MUFAJFK
        FOREIGN KEY (MUFAJ_ID)
        REFERENCES MUFAJ(ID)
        ON DELETE SET NULL
);


CREATE TABLE KESZLET
(
    "ARUHAZ_ID" NUMBER NOT NULL ENABLE, 
    "KONYV_ID" NUMBER NOT NULL ENABLE, 
    "MENNYISEG" NUMBER NOT NULL ENABLE, 
    CONSTRAINT KESZLET_ARUHAZFK 
        FOREIGN KEY (ARUHAZ_ID) 
        REFERENCES ARUHAZ(ID)
        ON DELETE CASCADE,
    CONSTRAINT KESZLET_KONYVFK
        FOREIGN KEY (KONYV_ID)
        REFERENCES KONYV(ID)
        ON DELETE CASCADE
);


CREATE TABLE VASARLAS
(
    "ID" NUMBER(5,0) NOT NULL ENABLE,
    "DATUM" TIMESTAMP(1) NOT NULL ENABLE,
    "FELHASZNALO_ID" NUMBER(5,0) NOT NULL ENABLE,
    "KONYV_ID" NUMBER NOT NULL ENABLE,
    "SZALLITASI_CIM" VARCHAR2(40 BYTE) NOT NULL ENABLE,
    "MENNYISEG" NUMBER NOT NULL ENABLE,
    
    PRIMARY KEY ("ID") ENABLE,
     
    CONSTRAINT VASARLAS_FELHASZNALOFK
        FOREIGN KEY (FELHASZNALO_ID)
        REFERENCES FELHASZNALO(ID)
        ON DELETE CASCADE,
    CONSTRAINT VASARLAS_KONYVFK
        FOREIGN KEY (KONYV_ID)
        REFERENCES KONYV(ID)
        ON DELETE CASCADE 
);

create or replace TRIGGER KIFUTO_TRIGGER
FOR INSERT OR UPDATE OF MENNYISEG ON "KESZLET"
COMPOUND TRIGGER

    TYPE t_stock IS TABLE OF KONYV.ID%TYPE INDEX BY PLS_INTEGER;
    v_books t_stock;
    v_book_id KONYV.ID%TYPE;

    AFTER EACH ROW IS
    BEGIN
        v_books(v_books.COUNT + 1) := :NEW.KONYV_ID;
    END AFTER EACH ROW;

    AFTER STATEMENT IS
        v_total_amount NUMBER;
    BEGIN
        FOR i IN 1 .. v_books.COUNT LOOP
            SELECT SUM(MENNYISEG) INTO v_total_amount
            FROM KESZLET
            WHERE KONYV_ID = v_books(i)
            GROUP BY KONYV_ID;

            IF v_total_amount <= 10 THEN
                UPDATE KONYV
                SET KIFUTO = 1
                WHERE ID = v_books(i);
            ELSE
                UPDATE KONYV
                SET KIFUTO = 0
                WHERE ID = v_books(i);
            END IF;
        END LOOP;
    END AFTER STATEMENT;

END KIFUTO_TRIGGER;




commit;



create or replace TRIGGER update_torzsvasarlo
AFTER INSERT ON "VASARLAS"
DECLARE
    CURSOR c_vasarlasok IS
        SELECT FELHASZNALO_ID, SUM(MENNYISEG) AS VASAROLT_DARAB
        FROM VASARLAS
        GROUP BY FELHASZNALO_ID;
BEGIN
    FOR vasarlas_rec IN c_vasarlasok
    LOOP
        IF vasarlas_rec.VASAROLT_DARAB >= 5 THEN
            UPDATE FELHASZNALO
            SET TORZSVASARLO = 1
            WHERE ID = vasarlas_rec.FELHASZNALO_ID;
        END IF;
    END LOOP;
END;



commit;


create or replace FUNCTION get_item_stock(p_item_id NUMBER, p_cursor OUT SYS_REFCURSOR)
    RETURN SYS_REFCURSOR IS
BEGIN
    OPEN p_cursor FOR
        SELECT aruhaz_id, a.cim AS aruhaz_cim, konyv_id, mennyiseg
        FROM keszlet k
        LEFT JOIN aruhaz a ON k.aruhaz_id = a.id
        WHERE konyv_id = p_item_id;

    RETURN p_cursor;
END get_item_stock;


commit;



create or replace FUNCTION update_keszlet(p_konyv_id NUMBER, p_mennyiseg NUMBER)
    RETURN BOOLEAN IS ret BOOLEAN;
BEGIN
    UPDATE KESZLET SET MENNYISEG = MENNYISEG - p_mennyiseg WHERE KONYV_ID = p_konyv_id and ROWNUM = 1;

    ret := TRUE;
    RETURN ret;
EXCEPTION
    WHEN NO_DATA_FOUND THEN
        ret := FALSE;
        RETURN ret;
END update_keszlet;


commit;
